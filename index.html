Simplified hooks


App.Adapter = DS.RESTAdapter.extend({
  extract: function(store, type, id, payload) {
    if (payload._summary) {
      payload.summaryDatas = payload._summary.mapProperty('_details');
      delete payload._summary;
    }
    return this._super(store, type, id, payload);
  }
});
